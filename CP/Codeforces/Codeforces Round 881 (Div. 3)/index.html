    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 55;

void solve() {

    int n; cin >> n;
    int a[maxn];
    for (int i = 1; i <= n; ++ i) cin >> a[i];

    sort(a + 1, a + n + 1);

    int ans = 0;
    for (int i = 1; i <= n / 2; ++ i)
        ans += a[n - i + 1] - a[i];

    cout << ans << "\n";
}

signed main() {

    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T = 1; cin >> T;
    while (T --) solve();

    return 0;
}

    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 2e5 + 5;

void solve() {

    int n; cin >> n;
    int a[maxn];
    for (int i = 1; i <= n; ++ i) cin >> a[i];

    int k = 0, p = 0;
    for (int i = 1; i <= n; ++ i)
        if (a[i] < 0 || !a[i] && p) p ++, a[i] = -a[i];
        else
            if (p) k ++, p = 0;
    if (p) k ++, p = 0;

    int sum = 0;
    for (int i = 1; i <= n; ++ i)
        sum += a[i];

    cout << sum << " " << k << "\n";
}

signed main() {

    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T = 1; cin >> T;
    while (T --) solve();

    return 0;
}

    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 2e5 + 5;

int n, q, cnt[maxn];
vector<int> g[maxn];

void dfs(int cur, int pre) {
    for (auto k : g[cur]) {
        if (k == pre) continue;
        dfs(k, cur);
        cnt[cur] += cnt[k];
    }
    cnt[cur] = (cnt[cur] ? cnt[cur] : 1);
}

void solve() {

    cin >> n;
    for (int i = 1; i <= n; ++ i)
        g[i].clear(), cnt[i] = 0;

    for (int i = 1; i <= n - 1; ++ i) {
        int x, y; cin >> x >> y;
        g[x].push_back(y);
        g[y].push_back(x);
    }

    dfs(1, -1);

    cin >> q;
    while (q --) {
        int a, b; cin >> a >> b;
        cout << cnt[a] * cnt[b] << "\n";
    }
}

signed main() {

    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T = 1; cin >> T;
    while (T --) solve();

    return 0;
}

    E. Tracking Segments¶

    binary search brute force data structures *1600

    题意¶

    给定 m 个区间，之后再给点 q 个互不相同的位置置 1，定义一个区间是美丽的：区间内 1 的数量严格大于 0 的数量。问最少几次操作后，可以保证至少有一个区间是美丽的

    思路¶

    二分 + 前缀和

    一开始考虑树状数组，以为要同时优化插入和查询两个过程，但是计算时间复杂度发现还是会超时，后来想到可以优化 q 次置 1 ，如果第 k 次置 1 已经有一个区间是美丽的，那么第 k + 1 次一定是满足条件的，所有 q 次置 1 是满足单调性的，可以用二分优化

    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 1e5 + 5;

int n, m, q, a[maxn], l[maxn], r[maxn], res[maxn], sum[maxn];

bool check(int x) {
//    memset(res, 0, sizeof res);
//    memset(sum, 0, sizeof sum);
    for (int i = 1; i <= n; ++ i)
        res[i] = 0, sum[i] = 0;
    for (int i = 1; i <= x; ++ i)
        res[a[i]] = 1;
    for (int i = 1; i <= n; ++ i)
        sum[i] = sum[i - 1] + res[i];
    for (int i = 1; i <= m; ++ i)
        if (2 * (sum[r[i]] - sum[l[i] - 1]) > r[i] - l[i] + 1)
            return 1;
    return 0;
}

void solve() {

    cin >> n >> m;
    memset(a, 0, sizeof a);
    for (int i = 1; i <= m; ++ i)
        cin >> l[i] >> r[i];

    cin >> q;
    for (int i = 1; i <= q; ++ i)
        cin >> a[i];

    int L = 0, R = q + 5;
    while (L + 1 != R) {
        int mid = L + R >> 1;
        if (check(mid)) R = mid;
        else L = mid;
    }

    cout << (check(R) ? R : -1) << "\n";
}

signed main() {

    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int T = 1; cin >> T;
    while (T --) solve();

    return 0;
}
