    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 55;

void solve() {

    char str[maxn]; scanf("%s", str + 1);

    bool ans = false;
    int len = strlen(str + 1);
    for (int i = 1; i <= len / 2 - 1; ++ i)
        if (str[i] != str[i + 1]) ans = true;

    printf("%s\n", ans ? "YES" : "NO");
}

signed main() {

    int T = 1; scanf("%lld", &T);
    while (T --) solve();

    return 0;
}

    B. Maximum Sum¶

    brute force sortings *1100

    这题贪不得呀贪不得

    题意¶

    对 n 个数执行以下两种操作之一 k 次，使得和最大

    • 删除两个最小元素
    • 删除一个最大元素

    思路¶

    枚举删除最大元素数量的所有可能性

    因为执行完 k 次删除操作，等价于在原排序数列的前端和后端各删除一串连续的元素，所以可以想到用前缀和预处理，在 O(1) 的时间得到一段区间的和

    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 2e5 + 5;

int a[maxn], s[maxn];

void solve() {

    int n, k; scanf("%lld %lld", &n, &k);

    for (int i = 1; i <= n; ++ i) scanf("%lld", a + i);
    sort(a + 1, a + n + 1);

    int sum = 0;
    memset(s, 0, sizeof s);
    for (int i = 1; i <= n; ++ i) {
        sum += a[i];
        s[i] = s[i - 1] + a[i];
    }

    int ans = 0;
    for (int i = 0; i <= k; ++ i)
        ans = max(ans, s[n - i] - s[2 * (k - i)]);

    printf("%lld\n", ans);
}

signed main() {

    int T = 1; scanf("%lld", &T);
    while (T --) solve();

    return 0;
}

    C. Contrast Value¶

    greedy implementation *1200

    题意¶

    给定 n 个数的数列，删除尽量多的数，满足删除后的数列相邻两数差的绝对值之和与删除前相同，输出操作后数列的大小

    思路¶

    找到数列中递增和递减的区间，保留头部的数，其余中间的数都可以删去（一个区间尾部的数算为下一个区间的头部），最后答案再加一即最后一个区间的尾部

    记得特判所有元素相等的情况

    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 3e5 + 5;

int a[maxn];

void solve() {

    int n; scanf("%lld", &n);
    for (int i = 1; i <= n; ++ i) scanf("%lld", a + i);

    bool tag = true;
    for (int i = 1; i <= n - 1; ++ i)
        if (a[i] != a[i + 1]) tag = false;
    if (tag) {
        printf("1\n");
        return;
    }

    int flag = 0;
    for (int i = 1; i <= n - 1; ++ i)
        if (a[i] != a[i + 1]) {
            flag = (a[i] < a[i + 1]);
            break;
        }

    int ans = n;
    for (int i = 1; i <= n - 1; ++ i) {
        if (a[i] <= a[i + 1] && flag == 1) ans --;
        else if (a[i] >= a[i + 1] && flag == 0) ans --;
        else flag ^= 1;
    }

    printf("%lld\n", ans + 1);
}

signed main() {

    int T = 1; scanf("%lld", &T);
    while (T --) solve();

    return 0;
}
