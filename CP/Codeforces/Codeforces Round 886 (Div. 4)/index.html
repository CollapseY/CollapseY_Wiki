    #include <bits/stdc++.h>
#define int long long
using namespace std;

void solve() {

    int a, b, c;
    scanf("%lld %lld %lld", &a, &b, &c);

    int res1 = a + b;
    int res2 = a + c;
    int res3 = b + c;

    bool ans = false;
    if (res1 >= 10 || res2 >= 10 || res3 >= 10) ans = true;

    printf("%s\n", ans ? "YES" : "NO");
}

signed main() {

    int T = 1; scanf("%lld", &T);
    while (T --) solve();

    return 0;
}

#define int long long
using namespace std;

void solve() {

    int n; scanf("%lld", &n);

    int ans = 0, mx = 0;
    for (int i = 1; i <= n; ++ i) {
        int a, b; scanf("%lld %lld", &a, &b);

}

    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 15;

void solve() {

    char str[maxn][maxn];
    for (int i = 1; i <= 8; ++ i)
        scanf("%s", str[i] + 1);

    int cnt = 0;
    char ans[100];
    for (int i = 1; i <= 8; ++ i)
        for (int j = 1; j <= 8; ++ j)
            if (str[i][j] != '.') ans[++ cnt] = str[i][j];

    for (int i = 1; i <= cnt; ++ i) printf("%c", ans[i]);
    printf("\n");
}

signed main() {

    int T = 1; scanf("%lld", &T);
    while (T --) solve();

    return 0;

    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 2e5 + 5;

int a[maxn];

void solve() {

    int n, k; scanf("%lld %lld", &n, &k);
    for (int i = 1; i <= n; ++ i) scanf("%lld", a + i);
    sort(a + 1, a + n + 1);

    int res = 0, tmp = 1;
    for (int i = 1; i <= n - 1; ++ i)
        if (a[i + 1] - a[i] <= k) tmp ++;
        else {
            res = max(res, tmp);
            tmp = 1;
        }

    res = max(res, tmp);
    int ans = n - res;
    printf("%lld\n", ans);
}

signed main() {

    int T = 1; scanf("%lld", &T);
    while (T --) solve();

    return 0;
}

    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 2e5 + 5;

int n, c, s[maxn];

__int128 getsum(int w) {
    __int128 res = 0;
    for (int i = 1; i <= n; ++ i)
        res += (2 * w + s[i]) * (2 * w + s[i]);
    return res;
}

void solve() {

    scanf("%lld %lld", &n, &c);
    for (int i = 1; i <= n; ++ i) scanf("%lld", s + i);

    int l = 0, r = 1e9 + 5;
    while (l + 1 != r) {
        int mid = (l + r) / 2;
        if (getsum(mid) >= c) r = mid;
        else l = mid;
    }

    printf("%lld\n", r);
}

signed main() {

    int T = 1; scanf("%lld", &T);
    while (T --) solve();

    return 0;
}

    F. We Were Both Children¶

    math number theory

    题意¶

    有 n 只青蛙，每只青蛙初始都在 0 位置，每秒会往前跳 a_i。你可以在位置 1 到 n 设置一个陷阱，陷阱会抓住经过它的所有青蛙，求最多能抓住多少青蛙

    思路¶

    在位置 i 放置陷阱能抓住所有跳跃距离是 i 的因数的青蛙，对所有跳跃距离在 n 以下的青蛙进行计数，枚举 1 到 n 的所有位置，每个位置分解因数

    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 2e5 + 5;

int a[maxn];

void solve() {

    int n; scanf("%lld", &n);
    for (int i = 1; i <= n; ++ i) scanf("%lld", a + i);

    unordered_map<int, int> cnt;
    for (int i = 1; i <= n; ++ i)
        if (a[i] <= n) cnt[a[i]] ++;

    int ans = 0;
    for (int i = 1; i <= n; ++ i) {
        int tmp = 0;
        for (int j = 1; j * j <= i; ++ j) {
            if (i % j == 0) {
                tmp += cnt[j];
                if (j * j != i) tmp += cnt[i / j];
            }
        }
        ans = max(ans, tmp);
    }

    printf("%lld\n", ans);
}

signed main() {

    int T = 1; scanf("%lld", &T);
    while (T --) solve();

    return 0;
}

    G. The Morning Star¶

    combinatorics data structures

    题意¶

    给出 n 个坐标，判断共有几对坐标满足东、南、西、北、东南、东北、西南、西北八个方向，一对坐标算两个答案

    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 2e5 + 5;

void solve() {

    int n; scanf("%lld", &n);

    map<int, int> mp1, mp2, mp3, mp4;
    while (n --) {
        int x, y; scanf("%lld %lld", &x, &y);
        mp1[x] ++, mp2[y] ++, mp3[x - y] ++, mp4[x + y] ++;
    }

    int ans = 0;
    for (auto &k : mp1)
        if (k.second != 1) ans += k.second * (k.second - 1);
    for (auto &k : mp2)
        if (k.second != 1) ans += k.second * (k.second - 1);
    for (auto &k : mp3)
        if (k.second != 1) ans += k.second * (k.second - 1);
    for (auto &k : mp4)
        if (k.second != 1) ans += k.second * (k.second - 1);

    printf("%lld\n", ans);
}

signed main() {

    int T = 1; scanf("%lld", &T);
    while (T --) solve();

    return 0;
}

    H. The Third Letter¶

    dsu

    题意¶

    有 n 个人，给出 m 对关系 a，b，d：a 在 b 前方 d 距离处（d < 0说明 a 在 b 后方），问能否存在一种方案能够同时满足这 m 对关系

    思路¶

    带权并查集 又双叒叕不记得了

    维护并查集的同时维护 a 距离 p[a] 的距离，遍历所有关系，满足则加入并查集，不满足输出 NO

    #include <bits/stdc++.h>
#define int long long
using namespace std;
const int maxn = 2e5 + 5;

int p[maxn], s[maxn];

int find(int x) {
    if (x != p[x]) {
        int tmp = p[x];
        p[x] = find(p[x]);
        s[x] += s[tmp];
    }
    return p[x];
}

void solve() {

    int n, m; scanf("%lld %lld", &n, &m);
    for (int i = 1; i <= n; ++ i)
        p[i] = i, s[i] = 0;

    bool ans = true;
    while (m --) {
        int a, b, d; scanf("%lld %lld %lld", &a, &b, &d);
        int pa = find(a), pb = find(b);
        if (pa == pb) {
            if (d != s[a] - s[b]) ans = false;
        } else {
            s[pa] = d - s[a];
            p[pa] = b;
        }
    }

    printf("%s\n", ans ? "YES" : "NO");
}

signed main() {

    int T = 1; scanf("%lld", &T);
    while (T --) solve();

    return 0;
}
